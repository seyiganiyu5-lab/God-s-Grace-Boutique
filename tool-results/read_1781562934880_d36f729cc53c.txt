   250→    e.stopPropagation()

   252→    else setCategoryDragOver(false)

   254→    if (file) handleImageUpload(file, target)
   255→  }

   257→  // Image upload handler
   258→  const handleImageUpload = async (file: File, target: 'product' | 'category') => {
   259→    if (!file) return
   260→    const setUploading = target === 'product' ? setProductUploading : setCategoryUploading
   261→    const setForm = target === 'product' ? setProductForm : setCategoryForm
   262→
   263→    // Validate - be permissive with types (some browsers don't set file.type properly)
   264→    if (file.type && !file.type.startsWith('image/')) {
   265→      toast.error(lang === 'fr' ? 'Type de fichier invalide. Seules les images sont autorisées.' : 'Invalid file type. Only images are allowed.')
   266→      return
   267→    }
   268→    if (file.size > 50 * 1024 * 1024) {
   269→      toast.error(lang === 'fr' ? 'Fichier trop volumineux. Maximum 50 Mo.' : 'File too large. Max 50MB.')
   270→      return
   271→    }
   272→
   273→    setUploading(true)
   274→    try {
   275→      const formData = new FormData()
   276→      formData.append('file', file)
   277→      const res = await fetch('/api/upload', { method: 'POST', body: formData })
   278→      if (res.ok) {
   279→        const data = await res.json()
   280→        if (data.url) {
   281→          setForm((prev: any) => ({ ...prev, image: data.url }))
   282→          toast.success(lang === 'fr' ? 'Image téléchargée avec succès' : 'Image uploaded successfully')
   283→        } else {
   284→          toast.error(data.error || (lang === 'fr' ? 'Échec du téléchargement' : 'Upload failed'))
   285→        }
   286→      } else {
   287→        let errorMsg = lang === 'fr' ? 'Échec du téléchargement' : 'Upload failed'
   288→        try {
   289→          const data = await res.json()
   290→          errorMsg = data.error || errorMsg
   291→        } catch {}
   292→        toast.error(errorMsg)
   293→      }
   294→    } catch {
   295→      toast.error(lang === 'fr' ? 'Échec du téléchargement — vérifiez votre connexion' : 'Upload failed — check your connection')
   296→    } finally {
   297→      setUploading(false)
   298→    }
   299→  }
