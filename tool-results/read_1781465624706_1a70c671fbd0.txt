   247→          setProductForm(f => ({ ...f, nameFr: data.translated }))

   294→      toast.error('Invalid file type. Only images are allowed.')
   295→      return

   298→      toast.error('File too large. Max 50MB.')
   299→      return
   300→    }
   301→
   302→    setUploading(true)
   303→    try {
   304→      const formData = new FormData()
   305→      formData.append('file', file)
   306→      const res = await fetch('/api/upload', { method: 'POST', body: formData })
   307→      if (res.ok) {
   308→        const data = await res.json()
   309→        if (data.url) {
   310→          setForm((prev: any) => ({ ...prev, image: data.url }))

   312→        } else {
   313→          toast.error(data.error || 'Upload failed')
   314→        }
   315→      } else {
   316→        let errorMsg = 'Upload failed'
   317→        try {
   318→          const data = await res.json()
   319→          errorMsg = data.error || errorMsg
   320→        } catch {}
   321→        toast.error(errorMsg)
   322→      }
   323→    } catch {
   324→      toast.error('Upload failed — check your connection')

   326→      setUploading(false)
