   297→      toast.error('File too large. Max 50MB.')
   298→      return

   301→    setUploading(true)
   302→    try {
   303→      const formData = new FormData()
   304→      formData.append('file', file)
   305→      const res = await fetch('/api/upload', { method: 'POST', body: formData })
   306→      if (res.ok) {
   307→        const data = await res.json()

   309→          setForm((prev: any) => ({ ...prev, image: data.url }))

   312→          toast.error(data.error || 'Upload failed')
   313→        }

   317→          const data = await res.json()

   319→        } catch {}
   320→        toast.error(errorMsg)
   321→      }
   322→    } catch {
   323→      toast.error('Upload failed — check your connection')
